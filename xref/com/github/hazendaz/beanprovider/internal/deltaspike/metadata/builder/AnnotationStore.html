23  package com.github.hazendaz.beanprovider.internal.deltaspike.metadata.builder;
24  
25  import static java.util.Collections.emptyMap;
26  import static java.util.Collections.emptySet;
27  import static java.util.Collections.unmodifiableSet;
28  
29  import java.lang.annotation.Annotation;
30  import java.util.Map;
31  import java.util.Set;
32  
33  /**
34   * A helper class used to hold annotations on a type or member.
35   */
36  class AnnotationStore {
37      private final Map<Class<? extends Annotation>, Annotation> annotationMap;
38      private final Set<Annotation> annotationSet;
39  
40      AnnotationStore(Map<Class<? extends Annotation>, Annotation> annotationMap, Set<Annotation> annotationSet) {
41          this.annotationMap = annotationMap;
42          this.annotationSet = unmodifiableSet(annotationSet);
43      }
44  
45      AnnotationStore() {
46          annotationMap = emptyMap();
47          annotationSet = emptySet();
48      }
49  
50      <T extends Annotation> T getAnnotation(Class<T> annotationType) {
51          return annotationType.cast(annotationMap.get(annotationType));
52      }
53  
54      Set<Annotation> getAnnotations() {
55          return annotationSet;
56      }
57  
58      boolean isAnnotationPresent(Class<? extends Annotation> annotationType) {
59          return annotationMap.containsKey(annotationType);
60      }
61  }
