6   
7   import static org.assertj.core.api.Assertions.assertThat;
8   import static org.assertj.core.api.Assertions.assertThatThrownBy;
9   
10  import com.github.hazendaz.beanprovider.internal.deltaspike.literal.AlternativeLiteral;
11  import com.github.hazendaz.beanprovider.internal.deltaspike.literal.NamedLiteral;
12  
13  import jakarta.enterprise.inject.Alternative;
14  import jakarta.inject.Named;
15  
16  import java.lang.annotation.Annotation;
17  import java.util.List;
18  
19  import org.junit.jupiter.api.Test;
20  
21  class AnnotationBuilderTest {
22  
23      @Deprecated
24      static class DeprecatedType {
25          // marker class
26      }
27  
28      @Test
29      void defaultAnnotationStoreIsEmpty() {
30          final AnnotationStore store = new AnnotationStore();
31          assertThat(store.getAnnotations()).isEmpty();
32          assertThat(store.isAnnotationPresent(Named.class)).isFalse();
33          assertThat(store.getAnnotation(Named.class)).isNull();
34      }
35  
36      @Test
37      void annotationBuilderSupportsAllMutationPaths() {
38          final AnnotationBuilder source = new AnnotationBuilder();
39          source.add(new NamedLiteral("source"));
40  
41          final AnnotationBuilder builder = new AnnotationBuilder();
42          builder.add(new AlternativeLiteral());
43          builder.addAll(List.<Annotation> of(new NamedLiteral("collection")));
44          builder.addAll(source.create());
45          builder.addAll(DeprecatedType.class);
46          builder.remove(Deprecated.class);
47  
48          assertThat(builder.isAnnotationPresent(Named.class)).isTrue();
49          assertThat(builder.getAnnotation(Named.class).value()).isEqualTo("source");
50          assertThat(builder.isAnnotationPresent(Alternative.class)).isTrue();
51          assertThat(builder.toString()).contains("source");
52      }
53  
54      @Test
55      void nullInputsAreRejected() {
56          final AnnotationBuilder builder = new AnnotationBuilder();
57          assertThatThrownBy(() -> builder.add(null)).isInstanceOf(IllegalArgumentException.class);
58          assertThatThrownBy(() -> builder.remove(null)).isInstanceOf(IllegalArgumentException.class);
59      }
60  }
